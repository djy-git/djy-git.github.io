- FUNCTION check()
> DO      Read a character and return if it's valid under some conditions
> INPUT   c: character
> OUTPUT  validness of INPUT: bool

CALL c ← convert(c)
IF st is empty THEN
    IF c % 2 = 0 THEN
        return False
    ELSE
        push c to st
        return True
top ← top element of st
IF c - top != 1 THEN
    IF c % 2 = 0 THEN
        return False
    ELSE
        push c to st
        return True
ELSE
    pop st
    return True

## Code
{% highlight python %}

class Solution:
    def __init__(self):
        self.st = []
        self.flag = False

    def isValid(self, s: str) -> bool:
        for i, c in enumerate(s):
            if not self.check(c, i):
                return False
        if len(self.st) != 0:
            return False
        return True

    def check(self, c, i):
        c = self.convert(c)
        if len(self.st) == 0:
            if c % 2 == 0:  # )
                return False
            else:  # (
                self.st.append(c)
                return True

        top = self.st[-1]  # 1
        if c - top != 1:
            if c % 2 == 0:  # ()
                return False
            else:  # ()
                self.st.append(c)  # 1
                return True
        else:
            self.st.pop()
            return True

    def convert(self, c):
        if c == '(':
            return 1
        elif c == ')':
            return 2
        elif c == '{':
            return 3
        elif c == '}':
            return 4
        elif c == '[':
            return 5
        elif c == ']':
            return 6
        exit(-1)
