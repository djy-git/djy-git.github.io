## Code
{% highlight python %}

class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        nums = sorted(nums1 + nums2)
        n = len(nums)
        n2 = n // 2
        if n % 2 is 0:
          return (nums[n2] + nums[n2 - 1]) / 2
        else:
          return nums[n2]
